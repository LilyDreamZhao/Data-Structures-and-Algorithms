#9. Palindrome Number  
#49%
class Solution(object):
    def isPalindrome(self, x):
        """
        :type x: int
        :rtype: bool
        """
        if x < 0 or  (x % 10 == 0 and x != 0):
            return False;
        
        revertedNumber = 0;
        while x > revertedNumber:
            revertedNumber = revertedNumber * 10 + x % 10;
            x /= 10;
        
        return x == revertedNumber or x == revertedNumber/10;#/10是剔除奇数个位时候产生的中间位数字
        
#9. Palindrome Number

class Solution(object):
    def isPalindrome(self, x):
        """
        :type x: int
        :rtype: bool
        """
        if x < 0:
            return False
        else:
            str_x = str(x)
        if str_x == str_x[::-1]:
            return True
        else:
            return False
